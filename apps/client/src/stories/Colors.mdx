import { Meta } from '@storybook/addon-docs/blocks';
import tokensCss from '../styles/tokens.generated.css?raw';

<Meta title='Foundation/Color' />

export const TOKEN_COLORS = Object.fromEntries(
  Array.from(tokensCss.matchAll(/--color-([a-z0-9-]+):\s*([^;]+);/g), ([, token, value]) => [token, value.trim()]),
);

export const getTokenColor = (token) => TOKEN_COLORS[token] ?? `var(--color-${token})`;

export const TokenPalette = ({ groups }) => (
  <div style={{ display: 'grid', gap: 24, margin: '16px 0 32px' }}>
    {groups.map((group) => (
      <section key={group.title}>
        <div style={{ marginBottom: 12 }}>
          <h3 style={{ margin: 0 }}>{group.title}</h3>
          {group.subtitle ? <p style={{ margin: '4px 0 0', color: '#737373' }}>{group.subtitle}</p> : null}
        </div>
        <div
          style={{
            display: 'grid',
            gridTemplateColumns: 'repeat(auto-fill, minmax(160px, 1fr))',
            gap: 12,
          }}
        >
          {group.tokens.map((token) => (
            <div
              key={token}
              style={{
                overflow: 'hidden',
                border: '1px solid #e8e8e8',
                borderRadius: 8,
                background: '#ffffff',
              }}
            >
              <div
                aria-label={token}
                style={{
                  height: 64,
                  borderBottom: '1px solid #e8e8e8',
                  background: getTokenColor(token),
                }}
              />
              <div style={{ padding: '8px 10px', fontSize: 13, fontWeight: 600 }}>{token}</div>
            </div>
          ))}
        </div>
      </section>
    ))}
  </div>
);

# Color

Veyor 클라이언트에서 사용하는 색상 토큰입니다.

토큰 소스는 `tokens/global.json`(core) · `tokens/semantic.json`(semantic) · `apps/client/tokens.json`(primitive) 이며, `pnpm tokens:build` 결과인 `src/styles/tokens.generated.css` 의 CSS 변수를 참조합니다. 피그마(Tokens Studio)에서 값을 수정해 푸시 → 빌드만 하면 이 문서도 자동으로 반영됩니다.

## Semantic

의미 기반 토큰입니다. 컴포넌트에서는 가능한 한 이 토큰을 사용합니다.
